二分查找利用了分治的思想，是一种初阶的查找方法，其总是利用范围来查找，就像综艺节目中猜商品价格，主持人总会对猜的价格给出“高了”或“低了”的提示，这样不断逼近目标值，最终可得到结果。

#### 以下是具体实现
```javascript?linenums
import java.util.Scanner;
public class BinaryChop {
	
	private static int Left;
	private static int Right;
	private static int index;
	private static int[] Object = new int[11];

	public static void main(String[] args) {
		Scanner scanner = new Scanner(System.in);
		System.out.println("Please input from small to big:");
		for(int i=1;i<Object.length;i++){
			if(scanner.hasNextInt()) Object[i] = scanner.nextInt();
		}
		if(scanner.hasNextInt()) index = scanner.nextInt();
		scanner.close();
		Left = 1;
		Right = Object.length - 1;
		new BinaryChop().Find();
	}
	
	public void Find(){
		if(index > Object[Right] || index < Object[Left])
			System.out.println("No element");
		else{
			int temp;
			while(Left < Right){
				temp = (Left+Right)/2;
				if(index < Object[temp]){
					Right = temp;
					if(index == Object[Right+1]){
						System.out.println("Object["+temp+"] = "+index);
						break;
					}
					if(index < Object[temp] && index > Object[temp-1]){
						System.out.println("No element");
						break;
					}
				}
				else if(index > Object[temp]){
					Left = temp;
					if(index == Object[Left+1]){
						System.out.println("Object["+temp+"] = "+index);
						break;
					}
					if(index > Object[temp] && index < Object[temp+1]){
						System.out.println("No element");
						break;
					}
				}
				else {
					System.out.println("Object["+temp+"] = "+index);
					break;
				}
			}
		}
	}

}
```
